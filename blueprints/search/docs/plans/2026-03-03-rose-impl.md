```go
// postings.go
package rose

// vbyteEncode appends the VByte encoding of v to buf and returns the result.
// Encoding: 7 bits per byte, LSB first; MSB=1 means more bytes follow.
func vbyteEncode(buf []byte, v uint32) []byte {
	for v >= 0x80 {
		buf = append(buf, byte(v&0x7F)|0x80)
		v >>= 7
	}
	return append(buf, byte(v))
}

// vbyteDecode reads a VByte-encoded uint32 from buf[pos:] and returns
// the decoded value and the new position after the encoded integer.
func vbyteDecode(buf []byte, pos int) (uint32, int) {
	var v uint32
	var shift uint
	for {
		b := buf[pos]
		pos++
		v |= uint32(b&0x7F) << shift
		if b < 0x80 {
			break
		}
		shift += 7
	}
	return v, pos
}
```

```go
const blockSize = 128

// packBlock encodes a block of postings into a compact binary format.
// docIDs are delta-encoded relative to blockBase (the last docID of the
// previous block, or 0 for the first block).
// Returns the encoded bytes and BlockMaxImpact.
func packBlock(docIDs []uint32, impacts []uint8, blockBase uint32) ([]byte, uint8) {
	if len(docIDs) == 0 {
		return nil, 0
	}
	var bmi uint8
	for _, imp := range impacts {
		if imp > bmi {
			bmi = imp
		}
	}

	// encode docID deltas
	var buf []byte
	prev := blockBase
	for _, id := range docIDs {
		buf = vbyteEncode(buf, id-prev)
		prev = id
	}
	// append impact scores as raw bytes
	buf = append(buf, impacts...)
	return buf, bmi
}

// unpackBlock decodes a packed block, returning docIDs and impacts.
// blockBase is the last docID of the previous block (0 for the first block).
// n is the number of postings in this block (stored separately in the segment).
func unpackBlock(data []byte, blockBase uint32, n int) ([]uint32, []uint8, error) {
	docIDs := make([]uint32, n)
	impacts := make([]uint8, n)

	pos := 0
	prev := blockBase
	for i := 0; i < n; i++ {
		if pos >= len(data) {
			return nil, nil, fmt.Errorf("block data too short at docID %d", i)
		}
		delta, newPos := vbyteDecode(data, pos)
		pos = newPos
		prev += delta
		docIDs[i] = prev
	}
	// impacts are stored as a raw byte array after all deltas
	if pos+n > len(data) {
		return nil, nil, fmt.Errorf("block data too short for impacts")
	}
	copy(impacts, data[pos:pos+n])
	return docIDs, impacts, nil
}
```

```go
// bm25.go
package rose

import "math"

const (
	bm25K1    = 1.2
	bm25B     = 0.75
	bm25Delta = 1.0
)

// idf computes the Robertson IDF (never negative).
//   IDF(t) = log( (N - df + 0.5) / (df + 0.5) + 1 )
// Reference: Kamphuis et al. ECIR 2020, §2 "Robertson" variant.
func idf(df, N uint32) float64 {
	n := float64(N)
	d := float64(df)
	return math.Log((n-d+0.5)/(d+0.5) + 1)
}

// bm25Plus computes the BM25+ score for one (term, document) pair.
//   tf     — raw term frequency in the document
//   df     — document frequency (number of docs containing the term)
//   dl     — document length (number of tokens)
//   avgdl  — average document length in the corpus
//   N      — total number of documents in the corpus
// Reference: spec/0646_rose_driver.md §2.1; RESEARCH.md §3.1.
func bm25Plus(tf, df, dl, avgdl, N uint32) float64 {
	if tf == 0 || df == 0 || N == 0 {
		return 0
	}
	idfScore := idf(df, N)
	norm := float64(dl) / float64(avgdl)
	tfF := float64(tf)
	tfNorm := (bm25K1 + 1) * tfF / (tfF + bm25K1*(1-bm25B+bm25B*norm))
	return idfScore*tfNorm + bm25Delta
}

// quantise converts a slice of float64 BM25+ scores into uint8 impact scores.
// Scores are normalised to [1, 255] relative to the max score in the list.
// Zero scores are clamped to 1 (BM25+ floor property).
// Reference: RESEARCH.md §3.2; Lu arXiv:2407.03618.
func quantise(scores []float64) []uint8 {
	if len(scores) == 0 {
		return nil
	}
	maxScore := 0.0
	for _, s := range scores {
		if s > maxScore {
			maxScore = s
		}
	}
	if maxScore == 0 {
		maxScore = 1
	}
	out := make([]uint8, len(scores))
	for i, s := range scores {
		q := int(math.Round(s / maxScore * 255))
		if q < 1 {
			q = 1
		}
		if q > 255 {
			q = 255
		}
		out[i] = uint8(q)
	}
	return out
}
```

```go
// analyze tokenises, lowercases, filters stopwords, and stems text.
// Returns a slice of stemmed tokens (may contain duplicates for TF counting).
func analyze(text string) []string {
	var tokens []string
	runes := []rune(text)
	inToken := false
	start := 0
	for i, r := range runes {
		isWord := unicode.IsLetter(r) || unicode.IsDigit(r)
		if isWord && !inToken {
			start, inToken = i, true
		} else if !isWord && inToken {
			inToken = false
			tok := string(runes[start:i])
			if t := processTok(tok); t != "" {
				tokens = append(tokens, t)
			}
		}
	}
	if inToken {
		tok := string(runes[start:])
		if t := processTok(tok); t != "" {
			tokens = append(tokens, t)
		}
	}
	return tokens
}

// analyzeQuery applies the same analysis pipeline as analyze. Identical
// processing ensures query terms match index terms exactly.
func analyzeQuery(text string) []string {
	return analyze(text)
}

// processTok lowercases, length-filters, stopword-filters, and stems one token.
// Returns "" if the token should be discarded.
func processTok(tok string) string {
	if len(tok) < 2 || len(tok) > 64 {
		return ""
	}
	// lowercase
	lower := ""
	for _, r := range tok {
		lower += string(unicode.ToLower(r))
	}
	// stopword filter
	if _, ok := stopwords[lower]; ok {
		return ""
	}
	// Snowball English stem
	stemmed := english.Stem(lower, false)
	if stemmed == "" || len(stemmed) < 2 {
		return ""
	}
	return stemmed
}
```

```go
// docstore.go
package rose

import (
	"encoding/binary"
	"fmt"
	"io"
	"os"
	"strings"
)

const docStoreMaxText = 512

// docEntry holds a loaded document store entry.
type docEntry struct {
	externalID string
	text       []byte
}

// docStore is an append-only sequential file storing document metadata.
// Layout per entry:
//   [4] ExternalIDLen uint32
//   [N] ExternalID bytes
//   [4] TextLen uint32
//   [M] Text bytes (up to docStoreMaxText)
type docStore struct {
	f       *os.File
	entries []docEntry // in-memory index; entry i = docIdx i
}

// openDocStore opens (or creates) the doc store at path, loading all
// existing entries into memory for O(1) access.
func openDocStore(path string) (*docStore, error) {
	f, err := os.OpenFile(path, os.O_RDWR|os.O_CREATE, 0o644)
	if err != nil {
		return nil, fmt.Errorf("docstore open %s: %w", path, err)
	}
	ds := &docStore{f: f}
	if err := ds.loadAll(); err != nil {
		f.Close()
		return nil, err
	}
	return ds, nil
}

// loadAll scans the file from the beginning to build the in-memory entry table.
func (ds *docStore) loadAll() error {
	if _, err := ds.f.Seek(0, io.SeekStart); err != nil {
		return err
	}
	var le = binary.LittleEndian
	hdr := make([]byte, 4)
	for {
		if _, err := io.ReadFull(ds.f, hdr); err != nil {
			if err == io.EOF || err == io.ErrUnexpectedEOF {
				break
			}
			return err
		}
		idLen := le.Uint32(hdr)
		idBytes := make([]byte, idLen)
		if _, err := io.ReadFull(ds.f, idBytes); err != nil {
			return err
		}
		if _, err := io.ReadFull(ds.f, hdr); err != nil {
			return err
		}
		textLen := le.Uint32(hdr)
		textBytes := make([]byte, textLen)
		if _, err := io.ReadFull(ds.f, textBytes); err != nil {
			return err
		}
		ds.entries = append(ds.entries, docEntry{
			externalID: string(idBytes),
			text:       textBytes,
		})
	}
	return nil
}

// append adds a new document to the store, returning its docIdx.
func (ds *docStore) append(externalID string, text []byte) (uint32, error) {
	if len(text) > docStoreMaxText {
		text = text[:docStoreMaxText]
	}
	// Seek to end
	if _, err := ds.f.Seek(0, io.SeekEnd); err != nil {
		return 0, err
	}

	var le = binary.LittleEndian
	var hdr [4]byte

	le.PutUint32(hdr[:], uint32(len(externalID)))
	if _, err := ds.f.Write(hdr[:]); err != nil {
		return 0, err
	}
	if _, err := ds.f.WriteString(externalID); err != nil {
		return 0, err
	}
	le.PutUint32(hdr[:], uint32(len(text)))
	if _, err := ds.f.Write(hdr[:]); err != nil {
		return 0, err
	}
	if _, err := ds.f.Write(text); err != nil {
		return 0, err
	}

	idx := uint32(len(ds.entries))
	ds.entries = append(ds.entries, docEntry{
		externalID: externalID,
		text:       text,
	})
	return idx, nil
}

// get returns the docEntry for the given internal docIdx.
func (ds *docStore) get(idx uint32) (docEntry, error) {
	if int(idx) >= len(ds.entries) {
		return docEntry{}, fmt.Errorf("docstore: idx %d out of range (%d entries)", idx, len(ds.entries))
	}
	return ds.entries[idx], nil
}

// close flushes and closes the underlying file.
func (ds *docStore) close() error {
	if ds.f == nil {
		return nil
	}
	err := ds.f.Close()
	ds.f = nil
	return err
}

// snippetFor returns a short text fragment containing one of the query terms.
// queryTerms should be pre-stemmed (as stored in the index).
func snippetFor(text []byte, queryTerms []string) string {
	words := strings.Fields(string(text))
	termSet := make(map[string]bool, len(queryTerms))
	for _, t := range queryTerms {
		termSet[t] = true
	}

	// Find index of first word that matches any query term (after stemming)
	hitIdx := -1
	for i, w := range words {
		stemmed := processTok(w)
		if termSet[stemmed] {
			hitIdx = i
			break
		}
	}

	if hitIdx < 0 {
		// No match: return first 20 words
		end := 20
		if end > len(words) {
			end = len(words)
		}
		return strings.Join(words[:end], " ")
	}

	start := hitIdx - 10
	if start < 0 {
		start = 0
	}
	end := hitIdx + 20
	if end > len(words) {
		end = len(words)
	}
	return strings.Join(words[start:end], " ")
}
```

```go
package rose

import (
	"context"
	"encoding/binary"
	"fmt"
	"os"
	"path/filepath"
	"sort"

// Magic bytes identifying a Rose segment file.
var segMagic = [4]byte{'R', 'O', 'S', 'E'}

const segVersion = 1

// termDictEntry is one entry in the segment's term dictionary.
type termDictEntry struct {
	Term           string
	DocFreq        uint32
	MaxImpact      uint8
	MaxImpactScore float32
	PostOffset     uint64 // byte offset from start of posting data region
	PostLen        uint32 // byte length of posting list data
	NumBlocks      uint32
	// BlockIndex: parallel arrays of block base docIDs and file offsets
	BlockBases   []uint32
	BlockOffsets []uint64 // offsets within the posting list data
}

// segment is an immutable on-disk posting list file.
type segment struct {
	path     string
	docCount uint32
	avgDocLen float32
	terms    []termDictEntry        // sorted by Term
	termMap  map[string]int        // term → index into terms
	// postBase is the absolute byte offset in the file where posting data begins
	postBase int64
}

// memPosting is one posting in the in-memory index.
type memPosting struct {
	docIdx uint32
	tf     uint16
}

// memDoc holds per-document metadata during in-memory accumulation.
type memDoc struct {
	externalID string
	text       []byte
	docLen     uint32 // total token count
}

// memIndex accumulates postings in memory before flushing.
type memIndex struct {
	postings map[string][]memPosting // term → postings
	docs     []memDoc
	sizeEst  int64
}

func newMemIndex() *memIndex {
	return &memIndex{postings: make(map[string][]memPosting)}
}

// addDoc is a helper for tests that bypasses full analysis.
func (mi *memIndex) addDoc(extID string, text []byte, terms []string, tfMap map[string]int) {
	docIdx := uint32(len(mi.docs))
	mi.docs = append(mi.docs, memDoc{
		externalID: extID,
		text:       text,
		docLen:     uint32(len(terms)),
	})
	for term, tf := range tfMap {
		mi.postings[term] = append(mi.postings[term], memPosting{docIdx: docIdx, tf: uint16(tf)})
		mi.sizeEst += 6
	}
}

// flushSegment writes mi to a new segment file in dir with the given sequence number.
// Returns the loaded segment for immediate use.
func flushSegment(_ context.Context, mi *memIndex, dir string, seq uint32) (*segment, error) {
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return nil, err
	}

	// Compute corpus stats
	N := uint32(len(mi.docs))
	var sumDocLen uint64
	for _, d := range mi.docs {
		sumDocLen += uint64(d.docLen)
	}
	avgdl := uint32(1)
	if N > 0 {
		avgdl = uint32(sumDocLen / uint64(N))
		if avgdl == 0 {
			avgdl = 1
		}
	}
	avgDocLen := float32(avgdl)

	// Sort terms
	terms := make([]string, 0, len(mi.postings))
	for t := range mi.postings {
		terms = append(terms, t)
	}
	sort.Strings(terms)

	// Build posting blocks for each term
	type termData struct {
		entry termDictEntry
		data  []byte // encoded posting list data
	}
	termDatas := make([]termData, 0, len(terms))

	for _, term := range terms {
		posts := mi.postings[term]
		// Sort by docIdx
		sort.Slice(posts, func(i, j int) bool { return posts[i].docIdx < posts[j].docIdx })

		df := uint32(len(posts))
		// Compute BM25+ scores
		scores := make([]float64, len(posts))
		for i, p := range posts {
			dl := mi.docs[p.docIdx].docLen
			if dl == 0 {
				dl = 1
			}
			scores[i] = bm25Plus(uint32(p.tf), df, dl, avgdl, N)
		}
		impacts := quantise(scores)

		// Find max score for dequantisation
		maxScore := 0.0
		for _, s := range scores {
			if s > maxScore {
				maxScore = s
			}
		}

		// Pack into blocks of 128
		var listData []byte
		var blockBases []uint32
		var blockOffsets []uint64
		var wholeListMax uint8
		blockBase := uint32(0)

		for i := 0; i < len(posts); i += blockSize {
			end := i + blockSize
			if end > len(posts) {
				end = len(posts)
			}
			slice := posts[i:end]
			impSlice := impacts[i:end]

			docIDs := make([]uint32, len(slice))
			for j, p := range slice {
				docIDs[j] = p.docIdx
			}

			blockOff := uint64(len(listData))
			blockBases = append(blockBases, blockBase)
			blockOffsets = append(blockOffsets, blockOff)

			blockData, bmi := packBlock(docIDs, impSlice, blockBase)
			if bmi > wholeListMax {
				wholeListMax = bmi
			}

			// Write block header: [bmi uint8][nPost uint8]
			listData = append(listData, bmi, uint8(len(slice)))
			listData = append(listData, blockData...)

			if len(docIDs) > 0 {
				blockBase = docIDs[len(docIDs)-1]
			}
		}

		termDatas = append(termDatas, termData{
			entry: termDictEntry{
				Term:           term,
				DocFreq:        df,
				MaxImpact:      wholeListMax,
				MaxImpactScore: float32(maxScore),
				PostLen:        uint32(len(listData)),
				NumBlocks:      uint32(len(blockBases)),
				BlockBases:     blockBases,
				BlockOffsets:   blockOffsets,
			},
			data: listData,
		})
	}

	// Write segment file: tmp → rename
	name := fmt.Sprintf("rose_%05d.seg", seq)
	tmpPath := filepath.Join(dir, name+".tmp")
	finalPath := filepath.Join(dir, name)

	f, err := os.Create(tmpPath)
	if err != nil {
		return nil, fmt.Errorf("create segment tmp: %w", err)
	}

	le := binary.LittleEndian

	// Header
	f.Write(segMagic[:])
	f.Write([]byte{segVersion})
	var buf4 [4]byte
	var buf8 [8]byte

	le.PutUint32(buf4[:], N)
	f.Write(buf4[:])
	// avgDocLen as float32
	bits := *(*uint32)((*[4]byte)(buf4[:]))
	_ = bits
	{
		b := make([]byte, 4)
		le.PutUint32(b, math_Float32bits(avgDocLen))
		f.Write(b)
	}
	le.PutUint32(buf4[:], uint32(len(termDatas)))
	f.Write(buf4[:])

	// Placeholder for PostingBase (we'll fill it after the dict)
	postBasePlaceholderOff, _ := f.Seek(0, 1) // current pos
	f.Write(buf8[:])                            // 8 bytes placeholder

	// Term dictionary (must record offsets after writing posting data)
	// First pass: compute posting data layout
	postOffset := uint64(0)
	for i := range termDatas {
		termDatas[i].entry.PostOffset = postOffset
		postOffset += uint64(termDatas[i].entry.PostLen)
	}

	// Write dict entries
	dictEnd, err := writeDictEntries(f, termDatas)
	if err != nil {
		f.Close()
		os.Remove(tmpPath)
		return nil, err
	}

	// Fill PostingBase
	postingBase := int64(dictEnd)
	f.Seek(postBasePlaceholderOff, 0)
	le.PutUint64(buf8[:], uint64(postingBase))
	f.Write(buf8[:])
	f.Seek(0, 2) // back to end

	// Write posting data
	for _, td := range termDatas {
		f.Write(td.data)
	}

	if err := f.Sync(); err != nil {
		f.Close()
		os.Remove(tmpPath)
		return nil, err
	}
	f.Close()

	if err := os.Rename(tmpPath, finalPath); err != nil {
		os.Remove(tmpPath)
		return nil, err
	}

	// Load and return the segment
	return openSegment(finalPath)
}

// termEntry looks up a term in the segment's sorted dictionary.
func (s *segment) termEntry(term string) (termDictEntry, bool) {

	if !ok {
		return termDictEntry{}, false

// readPostings decodes all postings for a term and returns parallel docID and impact slices.
func (s *segment) readPostings(term string) ([]uint32, []uint8, error) {
	entry, ok := s.termEntry(term)
	if !ok {
		return nil, nil, nil
	}
	// Read raw posting data
	f, err := os.Open(s.path)
	if err != nil {
		return nil, nil, err
	}
	defer f.Close()

	data := make([]byte, entry.PostLen)
	off := s.postBase + int64(entry.PostOffset)
	if _, err := f.ReadAt(data, off); err != nil {
		return nil, nil, fmt.Errorf("readPostings %s: %w", term, err)
	}

	// Decode blocks
	var allDocIDs []uint32
	var allImpacts []uint8
	blockBase := uint32(0)
	pos := 0
	for b := uint32(0); b < entry.NumBlocks; b++ {
		if pos+2 > len(data) {
			return nil, nil, fmt.Errorf("block header truncated at block %d", b)
		}
		// bmi := data[pos]   // not needed for full decode
		nPost := int(data[pos+1])
		pos += 2
		end := pos + entry.PostLen // safe upper bound; unpackBlock will stop at n
		if end > len(data) {
			end = len(data)
		}
		docIDs, impacts, err := unpackBlock(data[pos:end], blockBase, nPost)
		if err != nil {
			return nil, nil, fmt.Errorf("unpackBlock block %d: %w", b, err)
		}
		if len(docIDs) > 0 {
			blockBase = docIDs[len(docIDs)-1]
		}
		// Advance pos: decode each VByte to find how many bytes were used
		tmpBase := func() uint32 {
			if b == 0 {
				return 0
			}
			return entry.BlockBases[b]
		}()
		_ = tmpBase
		// Recompute consumed bytes via re-encoding (simple but correct)
		var consumed int
		prev := entry.BlockBases[b]
		for _, id := range docIDs {
			tmpBuf := vbyteEncode(nil, id-prev)
			consumed += len(tmpBuf)
			prev = id
		}
		consumed += nPost // impact bytes
		pos += consumed

		allDocIDs = append(allDocIDs, docIDs...)
		allImpacts = append(allImpacts, impacts...)
	}
	return allDocIDs, allImpacts, nil
}
```

```go
// wand.go
package rose

import "sort"

// wandHit is one result from the WAND top-k loop.
type wandHit struct {
	docID uint32
	score uint16 // accumulated uint8 impacts; uint16 prevents overflow (max 255 × 10 terms)
}

// listCursor tracks position in one term's decoded posting list.
// For simplicity, all postings are pre-decoded into allDocIDs/allImpacts.
// (A production implementation would decode block-by-block.)
type listCursor struct {
	allDocIDs  []uint32
	allImpacts []uint8
	pos        int // current position in allDocIDs
	// Block-max info (simplified: use whole-list max as block-max here;
	// real block-max is set from segment block headers)
	wholeListMax uint8
	blockMax     uint8 // max impact in current block (≤ wholeListMax)
}

func (c *listCursor) exhausted() bool {
	return c.pos >= len(c.allDocIDs)
}

func (c *listCursor) curDocID() uint32 {
	if c.exhausted() {
		return ^uint32(0) // sentinel: MaxUint32
	}
	return c.allDocIDs[c.pos]
}

func (c *listCursor) curImpact() uint8 {
	if c.exhausted() {
		return 0
	}
	return c.allImpacts[c.pos]
}

// advanceTo moves the cursor to the first docID >= target.
func (c *listCursor) advanceTo(target uint32) {
	for !c.exhausted() && c.allDocIDs[c.pos] < target {
		c.pos++
	}
}

// wandTopK returns the top-k (docID, score) pairs using Block-Max WAND.
// Reference: Grand et al. ECIR 2020; Ding & Suel WSDM 2013.
func wandTopK(cursors []*listCursor, k int) []wandHit {
	if len(cursors) == 0 || k <= 0 {
		return nil
	}

	// Min-heap of size k: tracks the k-th highest score seen.
	heap := make([]wandHit, 0, k)
	threshold := uint16(0)

	for {
		// 1. Remove exhausted cursors.
		active := cursors[:0]
		for _, c := range cursors {
			if !c.exhausted() {
				active = append(active, c)
			}
		}
		cursors = active
		if len(cursors) == 0 {
			break
		}

		// 2. Sort active cursors by current docID ascending.
		sort.Slice(cursors, func(i, j int) bool {
			return cursors[i].curDocID() < cursors[j].curDocID()
		})

		// 3. Find pivot: scan left to right, accumulate blockMax.
		//    Pivot = first index where prefix sum of blockMax >= threshold.
		//    We use wholeListMax as blockMax (simplified; full impl uses per-block max).
		var prefixSum uint16
		pivot := -1
		pivotDoc := uint32(0)
		for i, c := range cursors {
			prefixSum += uint16(c.wholeListMax)
			if prefixSum >= threshold+1 { // +1 because threshold = k-th score, we need strictly better
				pivot = i
				pivotDoc = c.curDocID()
				break
			}
		}
		if pivot < 0 {
			break // no pivot found → no more candidates can beat threshold
		}

		// 4. Check if leftmost cursor is at pivotDoc.
		if cursors[0].curDocID() == pivotDoc {
			// All cursors in [0..pivot] should be at pivotDoc.
			// Advance any that are behind (shouldn't happen if sorted, but be safe).
			for i := 0; i <= pivot; i++ {
				cursors[i].advanceTo(pivotDoc)
			}
			// Score this document: sum impacts from all cursors at pivotDoc.
			var score uint16
			for _, c := range cursors {
				if !c.exhausted() && c.curDocID() == pivotDoc {
					score += uint16(c.curImpact())
				}
			}
			// Push to heap.
			heap = heapPush(heap, wandHit{docID: pivotDoc, score: score}, k)
			if len(heap) == k {
				threshold = heap[0].score // min in heap
			}
			// Advance all cursors at pivotDoc.
			for _, c := range cursors {
				if !c.exhausted() && c.curDocID() == pivotDoc {
					c.pos++
				}
			}
		} else {
			// 5. Leftmost cursor is behind pivotDoc → advance it.
			cursors[0].advanceTo(pivotDoc)
		}
	}

	// Sort heap descending by score.
	sort.Slice(heap, func(i, j int) bool { return heap[i].score > heap[j].score })
	return heap
}

// heapPush inserts h into a min-heap of capacity k.
// If the heap is full and h.score > heap min, replaces the min.
func heapPush(heap []wandHit, h wandHit, k int) []wandHit {
	if len(heap) < k {
		heap = append(heap, h)
		heapifyUp(heap, len(heap)-1)
	} else if len(heap) > 0 && h.score > heap[0].score {
		heap[0] = h
		heapifyDown(heap, 0)
	}
	return heap
}

func heapifyUp(h []wandHit, i int) {
	for i > 0 {
		parent := (i - 1) / 2
		if h[parent].score > h[i].score {
			h[parent], h[i] = h[i], h[parent]
			i = parent
		} else {
			break
		}
	}
}

func heapifyDown(h []wandHit, i int) {
	n := len(h)
	for {
		smallest := i
		l, r := 2*i+1, 2*i+2
		if l < n && h[l].score < h[smallest].score {
			smallest = l
		}
		if r < n && h[r].score < h[smallest].score {
			smallest = r
		}
		if smallest == i {
			break
		}
		h[i], h[smallest] = h[smallest], h[i]
		i = smallest
	}
}
```

```go
// index.go
package rose

import (
	"context"
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"sync"
	"sync/atomic"

	"github.com/go-mizu/mizu/blueprints/search/pkg/index"
)

func init() {
	index.Register("rose", func() index.Engine { return &Engine{} })
}

// Engine is an embedded BM25+ FTS engine using VByte delta encoding,
// Block-Max WAND top-k retrieval, and LSM-style segment flushing.
// Reference: spec/0646_rose_driver.md; RESEARCH.md.
type Engine struct {
	dir       string
	mu        sync.Mutex
	mem       *memIndex // current in-memory accumulation buffer
	segments  []*segment
	docStore  *docStore
	meta      roseMeta
	nextSeq   uint32  // next segment sequence number
	totalDocs atomic.Int64
	mergeStop chan struct{}
	mergeDone chan struct{}

	// FlushThresholdMB controls when the in-memory buffer is flushed.
	// Default: 64 MB.
	FlushThresholdMB int64
}

type roseMeta struct {
	Version   int      `json:"version"`
	DocCount  int64    `json:"doc_count"`
	AvgDocLen float32  `json:"avg_doc_len"`
	Segments  []string `json:"segments"`
	NextSeq   uint32   `json:"next_seg_seq"`
}

func (e *Engine) Name() string { return "rose" }

func (e *Engine) Open(ctx context.Context, dir string) error {
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return fmt.Errorf("rose: mkdir %s: %w", dir, err)
	}
	e.dir = dir
	if e.FlushThresholdMB <= 0 {
		e.FlushThresholdMB = 64
	}

	// Open doc store
	ds, err := openDocStore(filepath.Join(dir, "rose.docs"))
	if err != nil {
		return fmt.Errorf("rose: open doc store: %w", err)
	}
	e.docStore = ds

	// Load metadata
	if err := e.loadMeta(); err != nil && !os.IsNotExist(err) {
		return fmt.Errorf("rose: load meta: %w", err)
	}

	// Open existing segments
	for _, name := range e.meta.Segments {
		seg, err := openSegment(filepath.Join(dir, name))
		if err != nil {
			return fmt.Errorf("rose: open segment %s: %w", name, err)
		}
		e.segments = append(e.segments, seg)
	}

	e.nextSeq = e.meta.NextSeq
	e.totalDocs.Store(e.meta.DocCount)
	e.mem = newMemIndex()

	// Start background merge goroutine
	e.mergeStop = make(chan struct{})
	e.mergeDone = make(chan struct{})
	go e.mergeLoop(e.mergeStop, e.mergeDone)

	return nil
}

func (e *Engine) Close() error {
	// Stop merge loop
	if e.mergeStop != nil {
		close(e.mergeStop)
		<-e.mergeDone
	}

	e.mu.Lock()
	defer e.mu.Unlock()

	// Flush any remaining in-memory postings
	if len(e.mem.docs) > 0 {
		seg, err := e.flushLocked(context.Background())
		if err != nil {
			return err
		}
		e.segments = append(e.segments, seg)
		e.mem = newMemIndex()
		if err := e.saveMeta(); err != nil {
			return err
		}
	}

	if e.docStore != nil {
		e.docStore.close()
	}
	return nil
}

func (e *Engine) Stats(ctx context.Context) (index.EngineStats, error) {
	e.mu.Lock()
	// Flush to count docs accurately
	if len(e.mem.docs) > 0 {
		seg, err := e.flushLocked(ctx)
		if err != nil {
			e.mu.Unlock()
			return index.EngineStats{}, err
		}
		e.segments = append(e.segments, seg)
		e.mem = newMemIndex()
		e.saveMeta()
	}
	e.mu.Unlock()

	return index.EngineStats{
		DocCount:  e.totalDocs.Load(),
		DiskBytes: index.DirSizeBytes(e.dir),
	}, nil
}

func (e *Engine) Index(ctx context.Context, docs []index.Document) error {
	if len(docs) == 0 {
		return nil
	}

	e.mu.Lock()
	defer e.mu.Unlock()

	for _, doc := range docs {
		tokens := analyze(string(doc.Text))
		if len(tokens) == 0 {
			continue
		}

		// Count term frequencies
		tfMap := make(map[string]int, len(tokens))
		for _, tok := range tokens {
			tfMap[tok]++
		}

		// Add to docstore
		docIdx, err := e.docStore.append(doc.DocID, doc.Text)
		if err != nil {
			return fmt.Errorf("rose: docstore append: %w", err)
		}

		// Add to memIndex
		memDocIdx := uint32(len(e.mem.docs))
		e.mem.docs = append(e.mem.docs, memDoc{
			externalID: doc.DocID,
			text:       doc.Text,
			docLen:     uint32(len(tokens)),
		})
		_ = docIdx // docStore and memIndex both track sequentially

		for term, tf := range tfMap {
			e.mem.postings[term] = append(e.mem.postings[term], memPosting{
				docIdx: memDocIdx,
				tf:     uint16(tf),
			})
			e.mem.sizeEst += 6
		}
		e.mem.sizeEst += int64(len(doc.Text) / 8) // rough text estimate
		e.totalDocs.Add(1)
	}

	// Flush if over threshold
	if e.mem.sizeEst >= e.FlushThresholdMB*1024*1024 {
		seg, err := e.flushLocked(ctx)
		if err != nil {
			return err
		}
		e.segments = append(e.segments, seg)
		e.mem = newMemIndex()
		return e.saveMeta()
	}
	return nil
}

// flushLocked flushes the current memIndex to a new segment.
// Caller must hold e.mu.
func (e *Engine) flushLocked(ctx context.Context) (*segment, error) {
	seq := e.nextSeq
	e.nextSeq++
	seg, err := flushSegment(ctx, e.mem, e.dir, seq)
	if err != nil {
		return nil, fmt.Errorf("rose: flush segment %d: %w", seq, err)
	}
	return seg, nil
}

func (e *Engine) Search(ctx context.Context, q index.Query) (index.Results, error) {
	limit := q.Limit
	if limit <= 0 {
		limit = 10
	}

	// Flush memIndex so we can search it too
	e.mu.Lock()
	var searchSegs []*segment
	if len(e.mem.docs) > 0 {
		seg, err := e.flushLocked(ctx)
		if err != nil {
			e.mu.Unlock()
			return index.Results{}, err
		}
		e.segments = append(e.segments, seg)
		e.mem = newMemIndex()
		e.saveMeta()
	}
	searchSegs = make([]*segment, len(e.segments))
	copy(searchSegs, e.segments)
	e.mu.Unlock()

	if len(searchSegs) == 0 {
		return index.Results{}, nil
	}

	// Analyse query
	queryTerms := analyzeQuery(q.Text)
	if len(queryTerms) == 0 {
		return index.Results{}, nil
	}
	// Deduplicate query terms
	seen := make(map[string]bool)
	uniqueTerms := queryTerms[:0]
	for _, t := range queryTerms {
		if !seen[t] {
			seen[t] = true
			uniqueTerms = append(uniqueTerms, t)
		}
	}
	queryTerms = uniqueTerms

	// Build one cursor per (segment × term) pair
	var cursors []*listCursor
	for _, seg := range searchSegs {
		for _, term := range queryTerms {
			docIDs, impacts, err := seg.readPostings(term)
			if err != nil || len(docIDs) == 0 {
				continue
			}
			// Map local docIDs to global docIDs via segmentBase
			// (simplified: use docIDs directly since each segment's
			// internal IDs are already globally unique per doc store idx)
			var wholeMax uint8
			for _, imp := range impacts {
				if imp > wholeMax {
					wholeMax = imp
				}
			}
			cursors = append(cursors, &listCursor{
				allDocIDs:    docIDs,
				allImpacts:   impacts,
				wholeListMax: wholeMax,
				blockMax:     wholeMax,
			})
		}
	}

	if len(cursors) == 0 {
		return index.Results{}, nil
	}

	// Run WAND
	hits := wandTopK(cursors, limit+q.Offset)

	// Apply offset, convert to index.Hit
	if q.Offset >= len(hits) {
		return index.Results{Total: len(hits)}, nil
	}
	hits = hits[q.Offset:]
	if len(hits) > limit {
		hits = hits[:limit]
	}

	results := index.Results{Total: len(hits)}
	for _, h := range hits {
		entry, err := e.docStore.get(h.docID)
		if err != nil {
			continue
		}
		snippet := snippetFor(entry.text, queryTerms)
		results.Hits = append(results.Hits, index.Hit{
			DocID:   entry.externalID,
			Score:   float64(h.score),
			Snippet: snippet,
		})
	}
	return results, nil
}

func (e *Engine) loadMeta() error {
	data, err := os.ReadFile(filepath.Join(e.dir, "rose.meta"))
	if err != nil {
		return err
	}
	return json.Unmarshal(data, &e.meta)
}

func (e *Engine) saveMeta() error {
	names := make([]string, len(e.segments))
	for i, seg := range e.segments {
		names[i] = filepath.Base(seg.path)
	}
	e.meta = roseMeta{
		Version:  1,
		DocCount: e.totalDocs.Load(),
		Segments: names,
		NextSeq:  e.nextSeq,
	}
	data, err := json.Marshal(e.meta)
	if err != nil {
		return err
	}
	tmp := filepath.Join(e.dir, "rose.meta.tmp")
	if err := os.WriteFile(tmp, data, 0o644); err != nil {
		return err
	}
	return os.Rename(tmp, filepath.Join(e.dir, "rose.meta"))
}

var _ index.Engine = (*Engine)(nil)
```

```go
// merge.go
package rose

import (
	"context"
	"sort"
	"time"
)

const (
	mergeMinSegments = 4
	mergeInterval    = 5 * time.Second
)

// mergeLoop is a background goroutine that periodically checks if a merge is needed.
func (e *Engine) mergeLoop(stop <-chan struct{}, done chan<- struct{}) {
	defer close(done)
	ticker := time.NewTicker(mergeInterval)
	defer ticker.Stop()
	for {
		select {
		case <-ticker.C:
			e.tryMerge(context.Background())
		case <-stop:
			return
		}
	}
}

// tryMerge merges segments if any tier has >= mergeMinSegments segments.
func (e *Engine) tryMerge(ctx context.Context) {
	e.mu.Lock()
	if len(e.segments) < mergeMinSegments {
		e.mu.Unlock()
		return
	}
	// Simple policy: merge all segments when there are >= mergeMinSegments.
	// A tiered policy would group by size; this keeps the implementation tractable.
	toMerge := make([]*segment, len(e.segments))
	copy(toMerge, e.segments)
	e.mu.Unlock()

	merged, err := mergeSegments(ctx, toMerge, e.dir, e.nextSeq)
	if err != nil {
		return // silently skip; will retry next tick
	}

	e.mu.Lock()
	e.nextSeq++
	// Replace merged segments with the new one
	e.segments = []*segment{merged}
	e.saveMeta()
	e.mu.Unlock()
}

// mergeSegments performs an N-way sorted merge of the given segments,
// writing a new segment file. Returns the new segment.
func mergeSegments(ctx context.Context, segs []*segment, dir string, seq uint32) (*segment, error) {
	if len(segs) == 0 {
		return nil, nil
	}
	if len(segs) == 1 {
		return segs[0], nil
	}

	// Collect all unique terms across all segments
	termSet := make(map[string]bool)
	for _, seg := range segs {
		for _, e := range seg.terms {
			termSet[e.Term] = true
		}
	}
	allTerms := make([]string, 0, len(termSet))
	for t := range termSet {
		allTerms = append(allTerms, t)
	}
	sort.Strings(allTerms)

	// Compute merged corpus stats
	var totalDocs, sumDocLen uint64
	for _, seg := range segs {
		totalDocs += uint64(seg.docCount)
		sumDocLen += uint64(float32(seg.docCount) * seg.avgDocLen)
	}
	mergedN := uint32(totalDocs)
	mergedAvgdl := uint32(1)
	if totalDocs > 0 {
		mergedAvgdl = uint32(sumDocLen / totalDocs)
		if mergedAvgdl == 0 {
			mergedAvgdl = 1
		}
	}

	// Build merged memIndex by reading all postings from all segments
	merged := newMemIndex()
	// Each segment gets a docIdx base offset
	bases := make([]uint32, len(segs))
	for i, seg := range segs {
		if i > 0 {
			bases[i] = bases[i-1] + segs[i-1].docCount
		}
		// Add placeholder docs
		for d := uint32(0); d < seg.docCount; d++ {
			merged.docs = append(merged.docs, memDoc{docLen: uint32(seg.avgDocLen)})
		}
	}

	// For each term, collect postings from all segments
	for _, term := range allTerms {
		for si, seg := range segs {
			docIDs, impacts, err := seg.readPostings(term)
			if err != nil || len(docIDs) == 0 {
				continue
			}
			base := bases[si]
			for i, docID := range docIDs {
				merged.postings[term] = append(merged.postings[term], memPosting{
					docIdx: base + docID,
					tf:     uint16(impacts[i]), // use quantised impact as proxy tf
				})
			}
		}
	}

	// Update corpus stats
	// (bm25Plus will use the merged N and avgdl)
	for i := range merged.docs {
		merged.docs[i].docLen = mergedAvgdl
	}
	_ = mergedN // used inside flushSegment via merged.docs length

	return flushSegment(ctx, merged, dir, seq)
}
```
